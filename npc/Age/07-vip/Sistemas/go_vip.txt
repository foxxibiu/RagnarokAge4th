			{
	case 1:
		if (getgroupid() == 5) {
		warp "s_room",50,73; // mude o mapa e as coordenadas da sua Sala VIP.
		}else {
		mes "Apenas jogadores VIP PREMIUM pode se teleportar para a Sala VIP.";
		}
		break;
	case 2:
		break;
	}
	
OnInit:
bindatcmd "govip",strnpcinfo(3)+"::OnAtcommand";
end;
}
